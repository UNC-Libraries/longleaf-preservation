def track_failure
  if @return_status.nil? || @return_status == 1
    @return_status = 1
  else
    @return_status = 2
  end
end

def track_status(status)
  if status == 2
    @return_status = 2
  elsif status == 0
    track_success
  elsif status == 1
    track_failure
  end
end
